const { defineConfig } = require('@vue/cli-service')
module.exports = defineConfig({
  publicPath: '/', // 默认是 '/'
  lintOnSave: process.env.NODE_ENV !== 'production', // 生产上 false，非生产上 true
  transpileDependencies: false, // 默认 false，忽略编译 node_modules
  productionSourceMap: false, // 不需要生产环境的 source map
  // configureWebpack 和 chainWebpack 均可新增/修改webpack配置，一般configureWebpack用新增，chainWebpack用修改。
  // configureWebpack: {
  //   // 选项写法
  //   module: {
  //     rules: [
  //       {
  //         test: /\.(png|jpe?g|gif|webp|avif)(\?.*)?$/,
  //         use: {}
  //       },
  //     ]
  //   },
      plugins: [
      // http://doc.ruoyi.vip/ruoyi-vue/other/faq.html#使用gzip解压缩静态文件
      new CompressionPlugin({
        cache: false,                                  // 不启用文件缓存
        test: /\.(js|css|html|jpe?g|png|gif|svg)?$/i,  // 压缩文件格式
        filename: '[path][base].gz[query]',            // 压缩后的文件名
        algorithm: 'gzip',                             // 使用gzip压缩
        minRatio: 0.8,                                 // 压缩比例，小于 80% 的文件不会被压缩
        deleteOriginalAssets: false                    // 压缩后删除原文件
      })
    ],
  // },
  configureWebpack: config => {
    // config.module.rules.push({
    //   test: /\.(png|jpe?g|gif|webp|avif)(\?.*)?$/,
    //   use: {}
    //   }
    // })
    if (process.env.NODE_ENV === 'production') {} else {}
  },
  chainWebpack: config => {
    // 只有函数写法（函数写法可判断环境）
    if(process.env.NODE_ENV === 'production'){
      // 【【去除生产环境的 console】】
      config.optimization.minimizer('terser').tap(args => {
        args[0].terserOptions.compress.drop_console = true
        return args
      })
    }
    // 【【设置图片转base64阈值】】
    config.module
    .rule('images')
      .set('parser', {
        dataUrlCondition: {
          maxSize: 4 * 1024 // 4KiB
        }
      })
  },
  // 更好的方式修改 css loader
  css: {
    loaderOptions: {
      css: {
        // 这里的选项会传递给 css-loader
      },
      postcss: {
        // 这里的选项会传递给 postcss-loader
      }
    }
  },
  // devServer的跨域处理：浏览器存在跨域，服务器不存在跨域，所以可通过服务器代理请求到目标服务器解决跨域。
  // fetch('/baidu/widget?id=LocalNews&ajax=json').then(res => res.json()).then(data => console.log(data));
  devServer: {
    proxy: {
      '/baidu': {
        target: 'https://news.baidu.com/',
        changeOrigin: true,
        pathRewrite: { '^/baidu': '' },
      }
    }
  }
  // http://localhost:8081/baidu/widget?id=LocalNews&ajax=json
  // https://news.baidu.com/baidu/widget?id=LocalNews&ajax=json
  // https://news.baidu.com/widget?id=LocalNews&ajax=json
})
  • 浏览器兼容处理：
    • css3兼容（加css前缀）： 修改.browserslistrc,会根据postcss-loader + autoprefixer + browserslist 给css3属性加前缀
    • es6语法兼容（通过babel）： 修改.browserslistrc,会根据babel-loader + @babel/preset-env或@vue/cli-plugin-babel/preset + browserslist 把es6语法转es5语法。 但某些非标准规范的语法如类的装饰器语法，不能通过@babel/preset-env解决，需要单独安装一些插件解决或者不使用该语法。
      bash
      # 修改 .browserslistrc
> 1%
last 2 versions
# not dead
not ie < 11 # 兼容ie11
    • es6内置api兼容（通过core-js或@babel/polyfill）： polyfill 用于处理js兼容性（对于@babel/preset-env不能处理的api等新特性如Promise）；webpack4中默认已经集成了polyfill，webpack5中为了提升打包速度和体积默认去除掉了polyfill，但是vue-cli集成了。
      js
      // babel.config.js
module.exports = {
  presets: [
    ['@vue/cli-plugin-babel/preset',
      {
        useBuiltIns: 'entry' // 这种方式会根据 browserslist 目标导入所有polyfill，少数兼容问题可能要特殊处理或者不使用该api
      }
    ]
  ]
}
//main.js
import 'core-js/stable'
import 'regenerator-runtime/runtime'
  • 浏览器兼容处理的现代模式：
    • vue-cli-service build --modern 生成一个dist包含两个版本，一个现代版的包，面向支持 ES modules 的现代浏览器，通过<script type="module">加载，另一个旧版的包，面向不支持的旧浏览器，通过<script nomodule>加载。
    • vue-cli5中默认就是现代模式，即vue-cli-service build 等同于 vue-cli-service build --modern，取消现代模式：vue-cli-service build --no-module。
