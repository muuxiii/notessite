vuex-3.x介绍 ​

vue2 适用 npm i vuex@3

vue3 适用 vuex@4，一般用 pinia

Vuex（Vue插件）能够集中管理（能读能改）多个组件共享的状态，是一种适用于任意组件间通信的方式，对不经常更新的数据进行缓存。

2. 基本使用：state、mutations、actions、getters ​

index.js

js
import Vue from 'vue'
import Vuex from 'vuex'
import {getTaskList} from "@/api/task";
// 源码：node_modules/vuex/dist/vuex.esm.browser.js
// Vuex是一个对象，Vue.use(Vuex) 会去调用 Vuex.install 方法，
// 执行 Vue.mixin({ beforeCreate: vuexInit }); 向每个组件混入一个beforeCreate钩子vuexInit；
// 每个组件在渲染时，执行 vuexInit 方法，往当前组件实例上挂载 $store 
// (在根实例上已挂载了$store，这样不用在每个组件实例上挂载$store了，直接拿根实例的$store挂载上，$store一致)
Vue.use(Vuex)
// state：公共状态
const state = {
  sum:0, // 计算总数（jsuan模块）
  supNum: 10, // 支持人数（vote模块）
  oppNum: 20, // 反对人数（vote模块）
}
// getters：公共计算属性
const getters = {
  // 只做了get劫持，不能直接修改；类似computed，有缓存。
  // 第二个参数是其他的计算属性
  ratio(state,getters) {
    let {supNum,oppNum} = state, total = supNum + oppNum;
    return total === 0 ? '---' : (supNum / total * 100).toFixed(2) + '%';
  },
  bigSum(state){   // 当state中的数据需要经过加工后再使用，或者说为了方便使用
		return state.sum * 10
	}
}
// mutations：同步修改state公共状态
const mutations = {
  INCREMENT(state,payload){
    state.sum += payload
  },
  DECREMENT(state,payload){
    state.sum -= payload
  },
  // 第二个是额外传参（只支持传一个参数，如果需要传多个参数，可用对象的形式传递）
  CHANGE(state,{type,step = 1}){
    if(type === 'sup'){
      state.oppNum += step // state 中的数据同组件中的data类似做了深度劫持
      return
    }
    if(type === 'opp'){
      state.oppNum += step
    }
  }
}
// actions：异步修改state公共状态（可先处理异步，然后只能通知mutations中的方法修改状态）
const actions = {
  incrementOdd(context,payload){
    if (context.state.sum % 2 === 1) {
      context.commit('INCREMENT',payload)
    }
  },
  incrementWait(context,payload){
    setTimeout(() => {
      context.commit('INCREMENT',payload)
    }, 1000);
  },
  // 第一个参数context是创建的Store实例，是精简后的 {state,getters,commit,dispatch,rootState,rootGetters}
  // 第二个是额外传参（只支持传一个参数，如果需要传多个参数，可用对象的形式传递）
  async change({commit},{type,step = 1}){
    try {
      await getTaskList({
        state: 2,
        pageNum: 1,
        pageSize: 10
      });
    }catch (e) {}
    commit('CHANGE',{type,step})
  }
}
//创建Store的实例
export default new Vuex.Store({
   state,
   mutations,
   actions,
   getters,
   modules: {}, // modules：模块化
   plugins: []  // plugins：插件
})

<script>

export default {
  data() {
    return {
      step: 1,
    };
  },
  // 组件中读取vuex中的数据（详见技巧章节）

    jia() { this.$store.commit('INCREMENT',this.step) },
    jian() { this.$store.commit('DECREMENT',this.step)},
    jiaOdd() { this.$store.dispatch('incrementOdd',this.step)},
    jiaWait() { this.$store.dispatch('incrementWait',this.step)},

    ...mapActions({click2: 'change'}),
    clickHandle(type){
      if(type === 'sup'){
        // this.$store.commit('CHANGE',{type})
        this.click1({type})
      }
      if(type === 'opp'){
        // this.$store.dispatch('change',{type})
        this.click2({type})
      }
    }
  },
};
</script>

3. 四个map方法的使用 ​

组件中引入：import {mapState, mapGetters, mapMutations, mapActions} from 'vuex'

  • mapState()：映射state中的数据为计算属性；
  • mapGetters()：映射getters中的数据为计算属性。
  • mapMutations()：生成与mutations对话的方法；
  • mapActions()：生成与actions对话的方法。

用法参考map方法的官方api文档

4. 模块化+命名空间 ​

让代码更好维护，让状态分类更加明确，用法参考map方法的官方api文档

index.js

import Vue from 'vue'
import Vuex, {createLogger} from 'vuex'
import jisuan from './modules/jisuan'
import vote from './modules/vote'

Vue.use(Vuex)

const plugins = []
// 开发环境下使用插件createLogger 
if(process.env.NODE_ENV !== 'production') {
  plugins.push(new createLogger())
}

// root 公共状态管理
const state = {
  sum: 0
}
// root 公共计算属性
const getters = {
  // 只做了get劫持，类似computed，有缓存，但不能直接修改。
  // 第一个参数state，包括root的和子模块的state {sum: 0, jisuan: {sum: 0}, vote: {...}}  做了set和get劫持
  // 第二个参数是计算属性：包括root的和子模块的计算属性 {newSum: 1, "vote/ratio": "33.33%"}  做了get劫持
  newSum(state, getters) {
    console.log('root getters',state,getters)
    return state.sum + 1;
  }
}
//root mutations
const mutations = {
  // 第一个参数 state 是总的
  setNum(state, num) {
    console.log('root mutations',state,num)
    state.sum = num
  }
}
//root actions
const actions = {
  // 第一个参数context，中的state、getters是总的
  setNumAsyn(context,value) {
    console.log('root actions',context)
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve(true)
        context.commit('setNum', value)
      },2000)
    })
  }
}

export default new Vuex.Store({
  // 第一部分：root 根状态管理
  state,
  getters,
  mutations,
  actions,
  modules: {   // 第二部分：模块中的状态管理 （键名为模块名）
    jisuan,
    vote
  },
  plugins
})

// 模块化后的store实例：

// 状态管理如下（root的状态直接挂载，模块里面的状态放在对象中，对象名是模块名），都已经做了get和set劫持

5. 持久化插件 ​

vuex中的状态在刷新浏览器后就会恢复成初始状态，需要将vuex中的状态进行持久化存储（localStroage sessionStroage cookie）等。

手动实现需要考虑在vuex的state中初始化状态时候从storage中读取，需要在mutations中改变状态时候存入storage。还需要考虑在读写的时候使用JSON去格式化storage数据。或许还需要考虑storage中是否存在某key，是否需要删除某key等。

借助插件 vuex-persistedstate 很容易实现

js
// store/index.js
import Vue from 'vue'
import Vuex from 'vuex'
import getters from './getters'
import app from './modules/app'
import settings from './modules/settings'
import user from './modules/user'
import createPersistedState from 'vuex-persistedstate' // 持久化插件
import createLogger from 'vuex/dist/logger' // 日志插件（见 4. 模块化+命名空间）

Vue.use(Vuex)

// 持久化存储（默认使用localStroage，默认持久化所有vuex的state，默认使用的键名是vuex）
// const storageState = createPersistedState()

const storageState = createPersistedState({
  // paths: ['settings.sidebarLogo']  // 持久化某个模块的某个状态 settings.sidebarLogo
  paths: ['settings', 'app']  // 持久化某几个模块
})

const sessionState = createPersistedState({
  storage: sessionStorage,  // 使用sessionStorage进行持久化
  paths: ['user']
})

const store = new Vuex.Store({
  modules: {
    app,
    settings,
    user
  },
  getters,
  // plugins: process.env.NODE_ENV !== 'production' ? [createLogger(), storageState, sessionState] : [storageState, sessionState]
  plugins: [storageState, sessionState] // 使用插件
})
export default store
