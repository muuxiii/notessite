vuex-3.x介绍 ​

vue2 适用 npm i vuex@3

Vuex能够集中管理（能读能改）多个组件共享的状态，是一个Vue插件，是一种适用于任意组件间通信的方式。

2. 基本使用：state、mutations、actions三个核心配置项 ​

index.js

js
import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)
//state对象——保存具体的数据
const state = {
  sum:0
}
//mutations对象——修改state中的数据
const mutations = {
   INCREMENT(state,value){
      state.sum += value
    },
    DECREMENT(state,value){
      state.sum -= value
    }
}
//actions对象——可以处理异步，然后调用mutations中的方法修改状态
const actions = {
   incrementOdd(context,value){
      if (context.state.sum % 2 === 1) {
        context.commit('INCREMENT',value)
      }
    },
    incrementWait(context,value){
      setTimeout(() => {
        context.commit('INCREMENT',value)
      }, 1000);
    }
}
//创建并暴露store
export default new Vuex.Store({
   state,
   mutations,
   actions
})

main.js

js
import store from './store'
new Vue({
  store,
})

组件中使用

html
<!--3.1 组件中读取vuex中的数据：$store.state.sum-->
<template>
  <div>
    <h1>当前求和为：{{ $store.state.sum }}</h1>
    <p>
      <select v-model.number="step">
        <option>1</option>
        <option>2</option>
        <option>3</option>
      </select>
      <button @click="jia">+</button>
      <button @click="jian">-</button>
      <button @click="jiaOdd">当前和为奇数再加</button>
      <button @click="jiaWait">等1s再加</button>
    </p>
  </div>
</template>
<!--3.2 组件中修改vuex中的数据：
$store.dispatch('action中的方法名',数据) 或
$store.commit('mutations中的方法名',数据)。
若没有网络请求或其他业务逻辑，组件中可以越过actions，即不写dispatch，直接编写commit
-->
<script>
export default {
  data() {
    return {
      step: 1,
    };
  },
  methods: {
    jia() {
      this.$store.commit('INCREMENT',this.step)
    },
    jian() {
      this.$store.commit('DECREMENT',this.step)
    },
    jiaOdd() {
      this.$store.dispatch('incrementOdd',this.step)
    },
    jiaWait() {
      this.$store.dispatch('incrementWait',this.step)
    },
  },
};
</script>

4. 四个map方法的使用 ​

组件中引入：import {mapState, mapGetters, mapMutations, mapActions} from 'vuex'

  • mapState()：映射state中的数据为计算属性；
  • mapGetters()：映射getters中的数据为计算属性。
js
computed: {
  //借助mapState生成计算属性：sum、school、student（对象写法）
  ...mapState({sum:'sum',school:'school',student:'student'}), // 值是在state中的状态，键是计算属性
  //借助mapState生成计算属性：sum、school、student（数组写法）
  ...mapState(['sum','school','student']),

  //借助mapGetters生成计算属性：bigSum（对象写法）
  ...mapGetters({bigSum:'bigSum'}),
  //借助mapGetters生成计算属性：bigSum（数组写法）
  ...mapGetters(['bigSum'])
}

js

  • mapMutations()：生成与mutations对话的方法；
  • mapActions()：生成与actions对话的方法。
js
methods:{

  1. 开启命名空间后，组件中读取state数据
js

持久化 ​

vuex中的状态在刷新浏览器后就会恢复成初始状态，需要将vuex中的状态进行持久化存储（localStroage sessionStroage cookie）等。

手动实现需要考虑在vuex的state中初始化状态时候从storage中读取，需要在mutations中改变状态时候存入storage。还需要考虑在读写的时候使用JSON去格式化storage数据。或许还需要考虑storage中是否存在某key，是否需要删除某key等。

借助插件 vuex-persistedstate 很容易实现

js
// store/index.js
import Vue from 'vue'
import Vuex from 'vuex'
import getters from './getters'
import app from './modules/app'
import settings from './modules/settings'
import user from './modules/user'
import createPersistedState from 'vuex-persistedstate' // 持久化插件
import createLogger from 'vuex/dist/logger' // 日志插件

Vue.use(Vuex)

// 持久化存储（默认使用localStroage，默认持久化所有vuex的state，默认使用的键名是vuex）
// const storageState = createPersistedState()

const storageState = createPersistedState({
  // paths: ['settings.sidebarLogo']  // 持久化某个模块的某个状态 settings.sidebarLogo
  paths: ['settings', 'app']  // 持久化某几个模块
})

const sessionState = createPersistedState({
  storage: sessionStorage,  // 使用sessionStorage进行持久化
  paths: ['user']
})

const store = new Vuex.Store({
  modules: {
    app,
    settings,
    user
  },
  getters,
  // plugins: process.env.NODE_ENV !== 'production' ? [createLogger(), storageState, sessionState] : [storageState, sessionState]
  plugins: [storageState, sessionState] // 使用插件
})
export default store
