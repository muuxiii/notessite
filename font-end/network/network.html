// 1. 引入express
const express = require('express')
// 2. 创建应用对象

// 3. 创建路由规则。 request是对请求报文的封装，response是对响应报文的封装

// 4. 监听端口，启动服务

  console.log(`服务已启动： http://localhost:3000`)

http协议-ajax请求 ​

http协议 ​

【请求报文】
请求行      GET /path?a=1\&b=2 HTTP/1.1        （请求方式 路径及参数 协议版本）
请求头      Cookie: xxxxxxxxxx                 （头的格式是键值对 键: 值）
            Content-Type: application/json
请求空行
请求体      username=admin\&password=admin123  （get请求请求体是空，post请求可以有请求体，浏览器查看 Form Data）

【响应报文】
响应行      HTTP/1.1 200 OK  （协议版本 响应状态码 响应状态字符串）
响应头      Access-Control-Allow-Credentials: true   （头的格式一致）
            Access-Control-Allow-Headers: Content-Type
            Access-Control-Allow-Methods: POST, GET
            Access-Control-Allow-Origin: https://www.baidu.com
            Content-Type: application/json
响应空行
响应体      <html> \<head>\</head> \<body>\</body> </html>
原生ajax请求示例
js
const express = require('express')
const app = express()

let id = 1
setInterval(()=>{
  id++
},3000)

// 1. GET.html
app.get('/server', (request, response) => {
  // 设置响应头 设置允许跨域
  response.setHeader('Access-Control-Allow-Origin', '*')
  // 设置响应体
  response.send('Hello World! GET')
})

// 2. POST.html
app.post('/server', (req, res) => {
  res.setHeader('Access-Control-Allow-Origin', '*')
  res.send('Hello World! POST')
})

// all表示可以接收任意类型的请求
app.all('/server', (req, res) => {
  res.setHeader('Access-Control-Allow-Origin', '*')
  // 设置响应头 允许自定义的请求头
  res.setHeader('Access-Control-Allow-Headers', '*')
  res.send('Hello World! POST')
})

// 3. 响应JSON.html 演示JSON格式的响应体
app.get('/server-json', (req, res) => {
  res.setHeader('Access-Control-Allow-Origin', '*')
  const data = { name: '张三', age: 14 }
  let str = JSON.stringify(data) // 将对象转换为json格式字符串，因为响应体格式只能是字符串
  res.send(str)
})

// 4. IE的get请求缓存问题.html
app.get('/ie', (req, res) => {
  res.setHeader('Access-Control-Allow-Origin', '*')
  res.send(`Hello IE ${id}`)
})

// 5. 网络超时与异常.html  6. 取消请求.html
app.get('/delay', (req, res) => {
  res.setHeader('Access-Control-Allow-Origin', '*')
  setTimeout(() => {
    res.send('Hello 延迟')
  }, 3000)
})

app.listen(3001, () => {
  console.log(`服务已启动： http://localhost:3001`)
})

  /**
   * ajax发送请求的基本操作 四步走
   */
  $('button').click(function () {
    // 1. 创建对象
    const xhr = new XMLHttpRequest()
    // 2. 设置请求信息
    xhr.open('GET', 'http://127.0.0.1:3001/server?a=1&b=2') //设置请求行：请求方式 请求url（get请求参数也是url的一部分）
    xhr.setRequestHeader('abc', '1234'); // 设置请求头（自定义）
    // 3. 发送请求
    xhr.send() // //get请求不传body参数，只有post请求传
    // 4. 接收响应（事件绑定，处理服务端返回的结果）
    xhr.onreadystatechange = function () {
      // onreadystatechange 的意思是 当readystate改变时触发（一共可触发4次：0 -> 1 -> 2 -> 3 -> 4）。
      // readystate有5个状态值 0(未初始化) 1(open方法调用完毕) 2(send方法调用完毕) 3(服务端返回部分结果) 4(服务端返回全部结果)
      if (xhr.readyState === 4) {
        // 判断响应状态码 200 400 401 403 500
        if (xhr.status >= 200 && xhr.status < 300) {
          // 处理响应结果 行 头 空行 体
          console.log('xhr', xhr);
          // 1. 响应行 (状态码，状态字符串)
          console.log(xhr.status, xhr.statusText);
          // 2. 响应头
          console.log(xhr.getAllResponseHeaders());
          // 3. 响应体
          console.log(xhr.response);
          // 我的需求：响应结果放在div中
          $('.result').html(xhr.response)
        }
      }
    }
  })
</script>

websocket协议-websocket请求 ​

长链接，双向通信，不受跨域限制，只能传递字符串

使用场景 ​

  • 场景一：数据的实时获取（图表数据的实时展示）
    • 采用ajax轮询（定时器，一直间隔发请求，效率低，非常浪费资源）
    • 采用websocket（数据改变后， 服务端主动向客户端发消息）
  • 场景二：即时通信（客服系统，直播，视频会议）

    一般使用第三方即时通信插件： 环信、腾讯云IM、360织语、声网

跨域问题 ​

http协议具有跨域问题，其他协议如：websocket 没有

TIP

产生跨域问题：浏览器的同源策略（浏览器的安全机制），当前源向另一个源发送http请求产生跨域（协议/域名/端口 不同）

解决方式： jsonp（script请求，不存在跨域，具有跨域能力的标签：img link iframe script） 、代理服务器（服务器请求不存在跨域）、cors响应头

代理服务 ​

webpack服务代理示例
js
// 请求： fetch('/baidu/widget?id=LocalNews&ajax=json').then(res => res.json()).then(data => console.log(data));

// webpack配置服务器代理
const { defineConfig } = require('@vue/cli-service')
module.exports = defineConfig({
  devServer: {
    proxy: {
      '/baidu': {
        target: 'https://news.baidu.com/',
        changeOrigin: true,
        pathRewrite: { '^/baidu': '' },
      }
    }
  }
})

// 浏览器请求 http://localhost:8081/baidu/widget?id=LocalNews&ajax=json 请求的是webpack的代理服务器（无跨域）
// 代理服务器匹配到特定的请求路径 '/baidu', 于是转发请求去请求目标服务（无跨域），
// 请求地址为目标源target+请求url https://news.baidu.com/baidu/widget?id=LocalNews&ajax=json
// '/baidu' => '' 于是代理服务的请求为 https://news.baidu.com/widget?id=LocalNews&ajax=json
// 代理服务器请求回来的数据再返回给浏览器

TIP

正向代理即是客户端代理, 代理客户端, 服务端不知道实际发起请求的客户端. 正向代理中，proxy和client同属一个LAN，对server透明.

反向代理即是服务端代理, 代理服务端, 客户端不知道实际提供服务的服务端. 反向代理中，proxy和server同属一个LAN，对client透明.

cors ​

服务端配置特殊的响应头 Access-Control-Allow-Origin

RESTful ​

get和post的区别 ​

  • 请求参数过多，或者请求参数大，用post
  • get请求参数会被保留在浏览器历史记录中，post不会保留，post相对get更安全。
  • get请求可以在书签中收藏，post不行
  • get请求会被浏览器主动缓存，post不会，除非手动设置
  • 浏览器回退时get是无害的，post在会再次提交请求
